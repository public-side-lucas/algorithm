## 💻 알고리즘 코드

```javascript
// 백준 스타일 입력 — require("fs").readFileSync 지원
const input = require("fs").readFileSync("/dev/stdin", "utf8").trim();
const lines = Number(input.split("\n")[0]);

function solve(n){
  let factor = 2

  while(factor <= Math.sqrt(n)){
    if(n%factor ===0) {
      console.log(factor)
      n /= factor
    }else{
      factor++
    }
  }
  if (n > 1) console.log(n); // 마지막에 남은 소수 출력
}

solve(lines)
```
