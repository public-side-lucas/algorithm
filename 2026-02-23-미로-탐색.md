function bfs(s,e){
  let queue = [[s,e,1]]
  let pointer = 0
  const position = [[1,0],[0,1],[-1,0],[0,-1]]

  while(queue[pointer]){
    const [start,end,count] = queue[pointer]

    for(let i =0; i<position.length; i++){
      const dx = start + position[i][0]
      const dy = end + position[i][1]

      if(dx === n-1 && dy === m-1)return count+1

      if(!matrix[dx]  || !matrix[dx][dy])continue

      matrix[dx][dy] = 0

      queue.push([dx,dy, count+1])
    }

    pointer++
  }
  

}
